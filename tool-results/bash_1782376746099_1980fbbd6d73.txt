/* ─── Hash-based router ─── */
function useHash() {
  const ash, setHash] = React.useState("");
  React.useEffect(() => {
    const h = window.location.hash.replace("#", "") || "hero";
    setHash(h);
    const onHash = () => setHash(window.location.hash.replace("#", "") || "hero");
    window.addEventListener("hashchange", onHash);
    return () => window.removeEventListener("hashchange", onHash);
  }, []);
  return hash;
}

function navigate(id: string) {
  window.location.hash = id;
  document.getElementById(id)?.scrollIntoView({ behavior: "smooth" });
}

/* ─── Sidebar data ─── */
function getSidebarData() {
  const staticLinks = [
    { name: "Introduction", slug: "hero" },
    { name: "Installation", slug: "installation" },
    { name: "Theming", slug: "theming" },
  ];
  const catLinks = categories.map((cat) => ({
    id: cat.id,
    label: cat.label,
    components: getComponentsByCategory(cat.id).map((c) => ({
      name: c.name.replace("Journal", ""),
      slug: c.name.replace("Journal", "").toLowerCase().replace(/\s+/g, "-"),
      fullName: c.name,
    })),
  }));
  return { staticLinks, catLinks };
}

/* ─── Component Doc Page (per component) ─── */
function ComponentDocPage({ entryKey }: { entryKey: string }) {
  const entry = registry[entryKey];
  if (!entry) return <p className="text-journal-sepia">Component not found.</p>;

  const installCmd = `bun run cli/install.ts ${entryKey}`;
  const componentName = entry.name;

  return (
    <section id={entry.name.replace("Journal", "").toLowerCase().replace(/\s+/g, "-")} className="scroll-mt-20 py-10 border-t border-journal-rule">
      <ComponentHeader name={componentName} description={entry.description} filename={entry.file} />

      <InstallCommand command={installCmd} />

      <h3 className="font-serif text-lg font-semibold text-journal-ink mt-10 mb-4">Preview</h3>
      <ComponentPreview>
        <ComponentPreviewDemo name={entryKey} />
      </ComponentPreview>

      <h3 className="font-serif text-lg font-semibold text-journal-ink mt-10 mb-4">Source Code</h3>
      <CodeBlock filename={entry.file}>{entry.code}</CodeBlock>

      <h3 className="font-serif text-lg font-semibold text-journal-ink mt-10 mb-2">Dependencies</h3>
      <p className="text-sm text-journal-ink-light font-serif mb-6">
        {entry.dependencies.length > 0
          ? entry.dependencies.map((d) => <code key={d} className="text-xs bg-journal-paper-dark px-1.5 py-0.5 rounded font-mono mr-2">{d}</code>)
          : <span className="text-journal-sepia italic">None — fully self-contained.</span>}
      </p>
    </section>
  );
}

/* ─── Tabs Demo (needs state) ─── */
function TabDemo() {
  const [active, setActive] = React.useState("prose");
  const tabs = [
    { id: "prose", label: "Prose", content: "Prose journaling is the most common form — a free-flowing narrative that captures events, thoughts, and emotions of the day." },
    { id: "poetry", label: "Poetry", content: "The poem in a journal occupies a unique space — it is neither fully public nor entirely private, a form of compressed observation." },
  ];
  const current = tabs.find((t) => t.id === active);
  return (
    <div className="border border-journal-rule rounded-sm overflow-hidden">
      <nav className="flex border-b border-journal-rule bg-journal-paper-dark/30" role="tablist">
        {tabs.map((t) => (
          <button key={t.id} role="tab" aria-selected={active === t.id} onClick={() => setActive(t.id)}
            className={`flex-1 px-4 py-2.5 text-xs uppercase tracking-wider font-medium transition-colors cursor-pointer ${active === t.id ? "text-journal-ink border-b-2 border-b-journal-burgundy bg-journal-paper" : "text-journal-sepia hover:text-journal-ink"}`}
          >{t.label}</button>
        ))}
      </nav>
      <div className="p-5 font-serif text-journal-ink-light leading-relaxed" role="tabpanel">{current?.content}</div>
    </div>
  );
}
